int Round(float a)
{
return (int) (a+0.5);
}
void lineDDA(void)
{
int dx=xend-xstart;
int dy=yend-ystart;
int steps,k;
float xincrement,yincrement,x=xstart,y=ystart;
if(abs(dx)>abs(dy))
{
steps=abs(dx);
}
else
steps=fabs(dy);
xincrement=(float) (dx)/(float) (steps);
yincrement=(float) (dy)/(float)(steps);
glBegin(GL_POINTS);
glVertex2i(Round(x),Round(y));
glEnd();
for(k=0;k < steps;k++)
{
x+=xincrement;
y+=yincrement;
glBegin(GL_POINTS);
glVertex2i(Round(x),Round(y));
glEnd();
}
glFlush();
}
